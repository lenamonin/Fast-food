    private void CreateButtonMouseClicked(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_CreateButtonMouseClicked
          
        String name = this.getNameValue();
        int accountNumber = this.getAccountNumber();
        
        if (!this.isNameValid(name)) {
            JOptionPane.showMessageDialog(this, "Le nom rentré n'est pas valide");
            return;
        }
        
        this.addNewLine(this.getNewLine(name, accountNumber), accountNumber);
        this.dispose();
    }//GEN-LAST:event_CreateButtonMouseClicked

    private void addNewLine(String newLine, int accountNumber) {
        try (BufferedWriter bw = new BufferedWriter(new FileWriter("data/comptes.txt", true))) {
            bw.write(newLine);
            JOptionPane.showMessageDialog(this, String.format("Compte créé avec succès !\nVotre numéro client est %d.", accountNumber));
        } catch (IOException e) {
            e.printStackTrace();
            JOptionPane.showMessageDialog(this, "Une erreur s'est produite");
        }
    }
