    @Override
    public void update(ArrayList<Order> orderList) {
        this.orderListPanel.removeAll();
        
        for (Order order: orderList) {
            this.orderListPanel.add(new OrderElement(order, this.mainKitchenUI));
        }
        
        this.revalidate();
        this.repaint();
    }
